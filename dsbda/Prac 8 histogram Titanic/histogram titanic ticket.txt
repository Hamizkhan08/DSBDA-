# Step 1: Import libraries
import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt

# Step 2: Load the Titanic dataset
df = pd.read_csv("Titanic.csv")  # Make sure Titanic.csv is in your current directory

# Step 3: View the first and last few rows
print(df.head())   # Shows first 5 rows
print(df.tail())   # Shows last 5 rows

# Step 4: Get descriptive statistics
print(df.describe())   # Shows mean, std, min, max for numeric columns

# Step 5: Count of passengers who survived vs not
sns.countplot(x="Survived", data=df)
plt.title("Survival Count (0 = No, 1 = Yes)")
plt.xlabel("Survived")
plt.ylabel("Count")
plt.show()

# Step 6: Barplot - Survival by gender
sns.barplot(x='Sex', y='Survived', data=df)
plt.title('Survival Based on Gender')
plt.xlabel('Gender')
plt.ylabel('Survival Rate')
plt.show()

# Step 7: Boxplot - Age distribution by survival
sns.boxplot(x='Survived', y='Age', data=df)
plt.title('Age Distribution by Survival')
plt.xlabel('Survived')
plt.ylabel('Age')
plt.show()

# Step 8: Histogram of ticket fares
sns.histplot(df['Fare'].dropna(), bins=30, kde=False)
plt.title('Distribution of Titanic Ticket Fares')
plt.xlabel('Fare')
plt.ylabel('Number of Passengers')
plt.show()
