import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

# Load the dataset
df = pd.read_csv("Titanic.csv")
df.columns = df.columns.str.strip()

# Ensure 'Age' and 'Survived' are numeric
df['Age'] = pd.to_numeric(df['Age'], errors='coerce')
df['Survived'] = pd.to_numeric(df['Survived'], errors='coerce')

# Drop rows with missing 'Age' or 'Survived'
df = df.dropna(subset=['Age', 'Survived'])

# Check required columns
if {'Sex', 'Age', 'Survived'}.issubset(df.columns):
    plt.figure(figsize=(10, 6))

    # Plot with distinct colors
    ax = sns.boxplot(x='Sex', y='Age', hue='Survived', data=df, palette="Set1")

    plt.xlabel("Gender")
    plt.ylabel("Age")
    plt.title("Age Distribution by Gender and Survival Status")

    # Custom legend labels
    handles, labels = ax.get_legend_handles_labels()
    plt.legend(handles=handles, labels=["Did Not Survive", "Survived"], title="Survival Status")

    plt.show()
else:
    print("One or more required columns ('Sex', 'Age', 'Survived') are missing in the dataset.")
